\documentclass[justified]{tufte-book}

\usepackage{pgffor} % \foreach
\usepackage{shortcuts}
\usepackage{lipsum}
\usepackage{concmath}

\newcommand*{\MinYear}{2021}
\newcommand*{\MaxYear}{2021}
\newcommand\TwoDigits[1]{\ifnum#1<10 0#1\else#1\fi}

\newcounter{day}[part]
\newcommand{\makedate}[3]{#3-#2-#1}
\newenvironment{logentry}[3]% Year, Month, Day, Heading
{\refstepcounter{day}\noindent\textbf{Day~\theday}\marginnote{\makedate{#1}{#2}{#3}}\par}{\vspace{0.5cm}}

\newcommand{\expect}[1]{\ensuremath{\mathbb{E}\left[#1\right]}}
\newcommand{\expectdist}[2]{\ensuremath{\mathbb{E}_{#1}\left[#2\right]}}

\begin{document}

\foreach \YYYY in {\MinYear,...,\MaxYear} {%
    \foreach \MM in {1,2,...,12} {%
        \foreach \DD in {1,2,...,31} {%
            \edef\FileName{\YYYY\TwoDigits{\MM}\TwoDigits{\DD}.lex}
            \IfFileExists{\FileName} {%
                \par
                \begin{logentry}{\YYYY}{\MM}{\DD}
                    \input{\FileName}%
                \end{logentry}
            } {%
                % File does not exist, don't do anything
            }%
        }%
    }%
}%

\end{document}
