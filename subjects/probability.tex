\chapter{Probability and Statistics}%
\label{cha:probability_and_statistics}

\begin{definition}[Unbiased Estimate]
    An estimator $\hat y$ is called an \emph{unbiased estimator} of $y$ if 
    \begin{equation}
        \bb{E}\left[\hat y \right] = y.
    \end{equation}
\end{definition}
